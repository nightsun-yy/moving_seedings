        window.showNodeDetail = function(type) {
            const panel = document.getElementById('topo-detail-panel');
            const titleEl = document.getElementById('topo-detail-title');
            const listEl = document.getElementById('topo-detail-list');
            
            const data = nodeDetailsData[type];
            if (!data) return;

            titleEl.innerHTML = `<i class="fa-solid fa-list-ul mr-2"></i>${data.title}`;
            listEl.innerHTML = ''; 

            const isBusOnline = linkStatus[type];

            data.items.forEach(item => {
                let isItemOnline = isBusOnline;
                let stateObj = null;
                
                if (type === 'sensors') {
                    isItemOnline = isBusOnline && sensorStates[item.id];
                    stateObj = sensorStates;
                } else if (type === 'motors') {
                    isItemOnline = isBusOnline && motorStates[item.id];
                    stateObj = motorStates;
                } else if (type === 'camera') {
                    isItemOnline = isBusOnline && cameraStates[item.id];
                    stateObj = cameraStates;
                }

                const statusColor = isItemOnline ? (currentMode==='virtual' ? 'text-green-400' : 'text-green-400') : 'text-red-500';
                const statusText = isItemOnline ? (currentMode==='virtual' ? '连接中(仿真)' : '在线') : '待机 / 断开';
                const statusIcon = isItemOnline ? 'fa-circle-check' : 'fa-circle-xmark';

                const li = document.createElement('li');
                li.className = 'flex items-center justify-between bg-slate-800 p-2 rounded border border-slate-700 text-xs';
                
                let actionBtnHtml = '';
                if (currentMode === 'virtual' && isBusOnline) {
                    const btnClass = stateObj[item.id] ? 'bg-slate-700 text-red-400 border border-red-900 hover:bg-slate-600' : 'bg-slate-700 text-green-400 border border-green-900 hover:bg-slate-600';
                    const btnText = stateObj[item.id] ? '<i class="fa-solid fa-power-off"></i> 断开' : '<i class="fa-solid fa-link"></i> 连接';
                    actionBtnHtml = `<button onclick="toggleSingleDevice('${item.id}', '${type}')" class="mt-1 px-1.5 py-0.5 rounded text-[9px] transition-colors shadow ${btnClass}">${btnText}</button>`;
                }

                li.innerHTML = `
                    <div class="flex items-center gap-2">
                        <div class="w-6 text-center text-slate-400"><i class="fa-solid ${item.icon}"></i></div>
                        <div>
                            <div class="text-slate-200 font-bold">${item.name}</div>
                            <div class="text-[10px] text-slate-500">${item.address}</div>
                        </div>
                    </div>
                    <div class="${statusColor} flex flex-col items-end">
                        <div class="flex items-center gap-1"><i class="fa-solid ${statusIcon}"></i> <span class="text-[9px] mt-0.5">${statusText}</span></div>
                        ${actionBtnHtml}
                    </div>
                `;
                listEl.appendChild(li);
            });

            panel.classList.remove('hidden');
        }

        window.toggleSingleDevice = function(deviceId, panelType) {
            let stateObj = null;
            if (panelType === 'sensors') stateObj = sensorStates;
            else if (panelType === 'motors') stateObj = motorStates;
            else if (panelType === 'camera') stateObj = cameraStates;

            if (stateObj) {
                stateObj[deviceId] = !stateObj[deviceId];
                addLog('参数配置', `虚拟模式下：已强制${stateObj[deviceId] ? '连接' : '断开'}设备 [${deviceId.toUpperCase()}]`, stateObj[deviceId] ? 'info' : 'warn');
                
                showNodeDetail(panelType); 
                
                if (panelType === 'sensors') updateSensorDataUI(); 
                if (panelType === 'motors') updateMotorControlUI();
                if (panelType === 'camera') updateCameraUI();
            }
        }

        function updateSensorDataUI() {
            const isBusOnline = linkStatus.sensors;
            
            const renderData = (id, value, unit) => {
                const el = document.getElementById(`val-${id}`);
                if (isBusOnline && sensorStates[id]) {
                    el.innerHTML = `${value}<span class="text-xs text-slate-500 ml-1">${unit}</span>`;
                } else {
                    el.innerHTML = `--<span class="text-xs text-slate-500 ml-1">${unit}</span>`;
                }
            };

            renderData('temp', (24 + Math.random()).toFixed(1), '℃');
            renderData('hum', (65 + Math.random() * 2).toFixed(1), '%');
            renderData('co2', Math.floor(450 + (Math.random() - 0.5) * 15), 'ppm');
            renderData('light', Math.floor(1200 + (Math.random() - 0.5) * 20), 'Lx');
            renderData('ec', (1.8 + (Math.random() - 0.5)*0.1).toFixed(2), 'mS');
            renderData('ph', (6.2 + (Math.random() - 0.5)*0.1).toFixed(2), 'pH');
        }

        function updateMotorControlUI() {
            const isBusOnline = linkStatus.motors;
            
            const sliderX = document.getElementById('slider-x');
            const sliderY = document.getElementById('slider-y');
            const btnRetract = document.getElementById('btn-cyl-retract');
            const btnExtend = document.getElementById('btn-cyl-extend');

            if (!isBusOnline || !motorStates.x) {
                sliderX.disabled = true;
                sliderX.classList.add('opacity-50', 'cursor-not-allowed');
            } else {
                sliderX.disabled = false;
                sliderX.classList.remove('opacity-50', 'cursor-not-allowed');
            }

            if (!isBusOnline || !motorStates.y) {
                sliderY.disabled = true;
                sliderY.classList.add('opacity-50', 'cursor-not-allowed');
            } else {
                sliderY.disabled = false;
                sliderY.classList.remove('opacity-50', 'cursor-not-allowed');
            }

            if(!isBusOnline || !motorStates.cyl){
                 btnRetract.classList.add('opacity-50', 'cursor-not-allowed');
                 btnExtend.classList.add('opacity-50', 'cursor-not-allowed');
            } else {
                 btnRetract.classList.remove('opacity-50', 'cursor-not-allowed');
                 btnExtend.classList.remove('opacity-50', 'cursor-not-allowed');
            }
        }

        function addLog(module, message, type='info') {
            const container = document.getElementById('log-container');
            const time = new Date().toLocaleTimeString('zh-CN', { hour12: false });
            
            let color = 'text-slate-300';
            if(type === 'error') color = 'text-red-400';
            if(type === 'success') color = 'text-green-400';
            if(type === 'warn') color = 'text-yellow-400';

            const logHtml = `<div class="mb-1"><span class="text-slate-500">[${time}]</span> <span class="text-blue-400">[${module}]</span> <span class="${color}">${message}</span></div>`;

            container.scrollTop = container.scrollHeight;

        window.moveToPosition = function(pos) {
            const sx = document.getElementById('slider-x');
            const sy = document.getElementById('slider-y');

            if(pos === 'top') { sy.value = 130; addLog('运动规划', '执行宏指令：前往顶层栽培架巡检'); }
            if(pos === 'home') { sx.value = 0; sy.value = 0; toggleCylinder(false); addLog('运动控制', '执行全局复位回零程序...', 'warn'); }
            
            updateKinetics();
        }

        function updateKinetics() {
            const vx = parseFloat(document.getElementById('slider-x').value);
            const vy = parseFloat(document.getElementById('slider-y').value);

            document.getElementById('ui-x-val').innerText = (vx > 0 ? '+' : '') + vx.toFixed(1) + ' mm';
            document.getElementById('ui-y-val').innerText = vy.toFixed(1) + ' mm';

            if (axisX_part && linkStatus.motors && motorStates.x) axisX_part.position.x = vx * 0.2; 
            if (axisY_part && linkStatus.motors && motorStates.y) axisY_part.position.y = vy * 0.2;
        }

        document.getElementById('slider-y').addEventListener('input', updateKinetics);
