    <!-- ==================== JavaScript 核心逻辑 ==================== -->
    <script>
        // --- 核心状态 ---
        let isVirtual = false;
        let isBusy = false;
        const target = { x: 0, y: 0, z: 0 };
        const current = { x: 0, y: 0, z: 0 };
        
        // 【关键配置】照片中的栽培架：底层空，中层有植物，顶层有植物
        const shelfPlants = [false, true, true]; 

        // --- 简易日志系统 ---
        const log = (msg, type="info") => {
            const box = document.getElementById('log-box');
            const div = document.createElement('div');
            const time = new Date().toLocaleTimeString('en-US', {hour12:false}).split(' ')[0];
            const color = type === 'success' ? 'text-emerald-400' : (type === 'error' ? 'text-red-400' : (type === 'warn' ? 'text-yellow-400' : 'text-slate-400'));
            div.innerHTML = `<span class="opacity-50">[${time}]</span> <span class="${color}">${msg}</span>`;
            box.appendChild(div);
            // 自动滚动到底部
            box.parentElement.scrollTop = box.parentElement.scrollHeight;
        };

        // ================== Three.js 场景与全景交互 ==================
        const container = document.getElementById('canvas-container');
        const scene = new THREE.Scene();
        
        // 增加网格地面，体现空间感和比例
        const gridHelper = new THREE.GridHelper(20, 20, 0x00d2ff, 0x1e293b);
        gridHelper.position.y = 0;
        // scene.add(gridHelper); // 可选：如果您觉得网格太乱可以注释掉这行

        // 调整摄像机视野，确保一打开就能看到全貌
        const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 100);
        // 最佳全景视角：稍远、稍高，看向中心区域
        camera.position.set(-1, 5, 11);
        
        const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        container.appendChild(renderer.domElement);

        // 引入 OrbitControls 轨道控制器，支持鼠标任意拖拽查看
        const controls = new THREE.OrbitControls(camera, renderer.domElement);
        controls.target.set(-1, 1.5, 0); // 旋转中心设为机器主体
        controls.enableDamping = true; // 平滑阻尼
        controls.maxPolarAngle = Math.PI / 2 + 0.1; // 限制不能看到地底下

        // 灯光布置，让金属更有质感
        scene.add(new THREE.AmbientLight(0xffffff, 0.6));
        const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);
        dirLight.position.set(5, 10, 5);
        scene.add(dirLight);
        const blueLight = new THREE.PointLight(0x00d2ff, 0.5, 20);
        blueLight.position.set(-3, 3, 4);
        scene.add(blueLight);

        // 材质定义
        const aluMat = new THREE.MeshStandardMaterial({ color: 0x9ca3af, metalness: 0.6, roughness: 0.3 }); // 铝型材
        const blackMat = new THREE.MeshStandardMaterial({ color: 0x111111, metalness: 0.2, roughness: 0.7 }); // 黑色底座
        const trayMat = new THREE.MeshStandardMaterial({ color: 0xe5e7eb, metalness: 0.5, roughness: 0.2 }); // 托盘

        // ================== 1:1 构建栽培架与植物 ==================
        const shelf = new THREE.Group();
        shelf.position.set(2, 0, 0); // 在右侧
        scene.add(shelf);

        // 立柱
        const addPost = (x, z) => {
            const p = new THREE.Mesh(new THREE.BoxGeometry(0.1, 4, 0.1), aluMat);
            p.position.set(x, 2, z);
            shelf.add(p);
        };
        addPost(-0.5, -1); addPost(1.5, -1); addPost(-0.5, 1); addPost(1.5, 1);

        // 每层高度及植物生成
        const plantsMeshes = [];
        const layerHeights = [0.6, 1.8, 3.0];
        
        layerHeights.forEach((y, i) => {
            // 托盘底座
            const tray = new THREE.Mesh(new THREE.BoxGeometry(1.6, 0.05, 1.8), trayMat);
            tray.position.set(0.5, y, 0);
            shelf.add(tray);

            if(shelfPlants[i]) {
                const plant = new THREE.Group();
                // 黑色花盆
                const pot = new THREE.Mesh(new THREE.BoxGeometry(1.0, 0.15, 1.2), blackMat);
                pot.position.y = 0.08;
                plant.add(pot);
                // 绿色植被
                for(let j=0; j<10; j++) {
                    const leaf = new THREE.Mesh(new THREE.BoxGeometry(0.2, 0.35, 0.2), new THREE.MeshStandardMaterial({color: 0x10b981, roughness:0.8}));
                    leaf.position.set((Math.random()-0.5)*0.7, 0.25, (Math.random()-0.5)*0.8);
                    leaf.rotation.set(Math.random(), Math.random(), Math.random());
                    plant.add(leaf);
                }
                plant.position.set(0.5, y, 0);
                shelf.add(plant);
                plantsMeshes[i] = plant;
            }
        });

        // ================== 1:1 构建执行机构 ==================
        const mechanism = new THREE.Group();
        scene.add(mechanism);

        // X轴导轨 (底层黑长条)
        const xRail = new THREE.Mesh(new THREE.BoxGeometry(7.5, 0.2, 0.4), blackMat);
        xRail.position.set(-1.5, 0.1, 0);
        mechanism.add(xRail);

        // X轴滑组
        const xGroup = new THREE.Group();
        mechanism.add(xGroup);
        const xSlide = new THREE.Mesh(new THREE.BoxGeometry(1.0, 0.1, 0.8), aluMat);
        xSlide.position.y = 0.25;
        xGroup.add(xSlide);

        // Y轴双立柱
        const yPillar = new THREE.Mesh(new THREE.BoxGeometry(0.2, 3.8, 0.2), aluMat);
        yPillar.position.set(0, 2.0, 0.2);
        const yPillar2 = new THREE.Mesh(new THREE.BoxGeometry(0.2, 3.8, 0.2), aluMat);
        yPillar2.position.set(0, 2.0, -0.2);
        xGroup.add(yPillar, yPillar2);

        // Y轴滑台
        const yGroup = new THREE.Group();
        xGroup.add(yGroup);
        const ySlide = new THREE.Mesh(new THREE.BoxGeometry(0.4, 0.5, 0.8), blackMat);
        const yPlate = new THREE.Mesh(new THREE.BoxGeometry(1.2, 0.1, 0.8), aluMat);
        yPlate.position.set(0.6, 0, 0);
        yGroup.add(ySlide, yPlate);

        // Z轴伸缩板
        const zGroup = new THREE.Group();
        yGroup.add(zGroup);
        const zPlate = new THREE.Mesh(new THREE.BoxGeometry(1.0, 0.05, 0.6), trayMat);
        zPlate.position.set(0.5, 0.08, 0);
        zGroup.add(zPlate);

        // ================== 渲染与物理插值 ==================
        function animate() {
            requestAnimationFrame(animate);
            controls.update(); // 允许鼠标拖拽

            // 平滑插值模拟工业惯性
            const speed = isVirtual ? 0.08 : 0.03;
            current.x += (target.x - current.x) * speed;
            current.y += (target.y - current.y) * speed;
            current.z += (target.z - current.z) * speed;

            // 百分比映射到物理空间尺寸
            xGroup.position.x = -4.5 + (current.x / 100) * 5.2;

        // ================== AI 寻苗与【有没有苗】判定反馈 ==================
        const wait = (ms) => new Promise(res => setTimeout(res, ms));
        
        async function runAutoPick() {
            if(isBusy) return;
            if(!isVirtual) {
                alert("安全规程提示：\n为防止硬件碰撞，算法流前置验证请先切换至【虚拟调试】模式！");
                return;
            }
            isBusy = true;
            log("下发指令：启动全自动寻苗验证序列", "success");

            // UI 初始化
            [1, 2, 3].forEach(id => {
                const el = document.getElementById(`slot-${id}`);
                el.innerText = "等待扫描...";
                el.className = "text-[10px] text-slate-500";
            });

            // 回位
            target.x = 5; target.y = 5; target.z = 0;
            await wait(1500);

            // 巡检点 (底层0, 中层1, 顶层2)
            const steps = [
                { id: 1, name: "底层槽位", y: 8, has: shelfPlants[0] },
                { id: 2, name: "中层槽位", y: 48, has: shelfPlants[1] },
            ];

            for(const step of steps) {
                log(`[轴控制] 前往 ${step.name}...`);
                target.x = 90; target.y = step.y; target.z = 0;
                await wait(2000);
                
                // 开始视觉扫描
                log(`[视觉流] 开启摄像头，分析目标...`);
                document.getElementById('ai-status').innerText = "扫描中";
                document.getElementById('ai-box').style.borderColor = "rgba(234,179,8,0.6)"; // Yellow
                document.getElementById(`slot-${step.id}`).innerText = "分析中...";
                document.getElementById(`slot-${step.id}`).className = "text-[10px] text-yellow-400 animate-pulse";
                
                target.z = 20; // 伸头微探
                await wait(1500);

                if(!step.has) {
                    // 没有苗
                    log(`[分析结果] ${step.name} -> 空置`, "error");
                    document.getElementById('ai-status').innerText = "空置槽";
                    document.getElementById('ai-box').style.borderColor = "rgba(239,68,68,0.8)"; // Red
                    
                    const slotUI = document.getElementById(`slot-${step.id}`);
                    slotUI.innerText = "❌ 空置 / 无植物";
                    slotUI.className = "text-[10px] text-red-400 font-bold bg-red-500/10 px-2 py-0.5 rounded";
                    
                    target.z = 0;
                    await wait(1000);
                } else {
                    // 发现苗！
                    log(`[分析结果] ${step.name} -> 发现合格植株！准备抓取`, "success");
                    document.getElementById('ai-status').innerText = "锁定目标";
                    document.getElementById('ai-box').style.borderColor = "#00ff41"; // Green
                    document.getElementById('ai-label').style.display = "block";

                    const slotUI = document.getElementById(`slot-${step.id}`);
                    slotUI.innerText = "✅ 发现幼苗";
                    slotUI.className = "text-[10px] text-emerald-400 font-bold bg-emerald-500/10 px-2 py-0.5 rounded shadow-[0_0_10px_rgba(16,185,129,0.3)]";
                    
                    // 抓取动作
                    target.z = 95;
                    await wait(1500);
                    
                    // 挂载拿走
                    zGroup.attach(plantsMeshes[step.id-1]); 
                    log("[轴控制] 移载完成，退出架子...");
                    
                    target.z = 0;
                    await wait(1500);
                    
                    log("[轴控制] 返回安全下料点...");
                    target.x = 10; target.y = 30;
                    await wait(2000);
                    break; // 抓到即停止
                }
            }

            log("【前置验证结束】控制闭环验证成功", "success");
            document.getElementById('ai-status').innerText = "待命";
            document.getElementById('ai-box').style.borderColor = "transparent";
            document.getElementById('ai-label').style.display = "none";
            isBusy = false;
        }

        document.getElementById('btn-ai-run').onclick = runAutoPick;
